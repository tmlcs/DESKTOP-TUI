86 // Write styled text at position
87 void write(int x, int y, const std::string& text, const Style& style) {
88 if (y >= rows_ || y < -1000) return; // completely off-screen
89 int col = x;
90 int row = y;
91
92 const char* p = text.data();
93 const char* end = p + text.size();
94 while (p < end) {
95 char32_t ch = utf8_decode(p, end);
96 if (ch == 0 && p >= end) break;
97
98 if (ch == '\n') {
99 col = x;
100 row++;
101 continue;
102 }
103
104 // Determine display width for column advancement
105 int cell_width = 0;
106 if (ch < 0x20 || ch == 0x7F) {
107 cell_width = 0; // Control chars: no display width
108 } else if (is_wide_codepoint(ch)) {
109 cell_width = 2; // CJK/wide: 2 terminal cells
110 } else {
111 cell_width = 1; // Normal: 1 cell
112 }
113
114 // Write to the primary cell if within bounds
115 if (col >= 0 && row >= 0 && col < cols_ && row < rows_) {
116 Cell& cell = back_buffer_[row * cols_ + col];
117 cell.ch = ch;
118 cell.style = style;
119 dirty_ = true;
120 // Clear the second cell for wide characters to prevent garbage
121 if (cell_width == 2 && col + 1 < cols_) {
122 Cell& cell2 = back_buffer_[row * cols_ + col + 1];
123 cell2.ch = ' ';
124 cell2.style = style;
125 }
126 }
127
128 col += cell_width;
129
130 // Handle line wrap if character extends past end
131 if (col >= cols_) {
132 col = x;
133 row++;
134 }
135 }
136 }

242 // Flush dirty regions to terminal
243 void flush() {
244 if (!dirty_) return;
245
246 // Full screen redraw (simplified; a more advanced version would track
247 // per-row dirty rects and use cursor positioning optimization)
248 term_.cursor_hide();
249
250 Style current_style = Style::Default();
251 bool style_set = false;
252
253 for (int row = 0; row < rows_; row++) {
254 bool row_changed = false;
255 for (int col = 0; col < cols_; col++) {
256 const Cell& back = back_buffer_[row * cols_ + col];
257 const Cell& front = front_buffer_[row * cols_ + col];
258 if (back != front) {
259 row_changed = true;
260 break;
261 }
262 }
263 if (!row_changed) continue;
264
265 // Move cursor to start of row
266 term_.cursor_move(0, row);
267
268 // Write entire row with style runs
269 int col = 0;
270 while (col < cols_) {
271 const Cell& cell = back_buffer_[row * cols_ + col];
272
273 // Find run of same style
274 int run_end = col + 1;
275 while (run_end < cols_ &&
276 back_buffer_[row * cols_ + run_end].style == cell.style) {
277 run_end++;
278 }
279
280 // Emit style change
281 if (!style_set || !(cell.style == current_style)) {
282 emit_style(cell.style);
283 current_style = cell.style;
284 style_set = true;
285 }
286
287 // Write the run
288 std::string run;
289 for (int i = col; i < run_end; i++) {
290 run += utf8_encode(back_buffer_[row * cols_ + i].ch);
291 }
292 term_.write(run);
293
294 // Update front buffer
295 for (int i = col; i < run_end; i++) {
296 front_buffer_[row * cols_ + i] = back_buffer_[row * cols_ + i];
297 }
298
299 col = run_end;
300 }
301 }
302
303 term_.write("\033[0m");
304 term_.cursor_show();
305 term_.flush();
306
307 dirty_ = false;
308 }
