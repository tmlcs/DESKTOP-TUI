14inline char32_t utf8_decode(const char*& p, const char* end) {
15 if (p >= end) return 0;
16 unsigned char b0 = static_cast<unsigned char>(*p);
17 if (b0 < 0x80) { p++; return b0; }
18 if (b0 < 0xC0) { p++; return 0; } // continuation byte alone
19 int extra = 0;
20 if (b0 < 0xE0) { extra = 1; b0 &= 0x1F; }
21 else if (b0 < 0xF0) { extra = 2; b0 &= 0x0F; }
22 else { extra = 3; b0 &= 0x07; }
23 char32_t cp = b0;
24 for (int i = 0; i < extra; i++) {
25 p++;
26 if (p >= end) return 0;
27 unsigned char b = static_cast<unsigned char>(*p);
28 if ((b & 0xC0) != 0x80) return 0;
29 cp = (cp << 6) | (b & 0x3F);
30 }
31 p++;
32 return cp;
33}

58inline bool is_wide_codepoint(char32_t ch) {
59 // Binary search through the sorted ranges
60 int left = 0;
61 int right = static_cast<int>(sizeof(wide_ranges) / sizeof(wide_ranges[0])) - 1;
62
63 while (left <= right) {
64 int mid = left + (right - left) / 2;
65 if (ch < wide_ranges[mid].first) {
66 right = mid - 1;
67 } else if (ch > wide_ranges[mid].second) {
68 left = mid + 1;
69 } else {
70 return true; // Found range containing ch
71 }
72 }
73 return false;
74}

78inline size_t display_width(const std::string& utf8_str) {
79 const char* p = utf8_str.data();
80 const char* end = p + utf8_str.size();
81 size_t width = 0;
82 while (p < end) {
83 char32_t ch = utf8_decode(p, end);
84 if (ch == 0 && p >= end) break;
85 if (is_wide_codepoint(ch)) {
86 width += 2;

99inline std::string truncate(const std::string& utf8_str, size_t max_width) {
100 size_t dw = display_width(utf8_str);
101 if (dw <= max_width) return utf8_str;
102
103 const char* p = utf8_str.data();
104 const char* end = p + utf8_str.size();
105 size_t width = 0;
106 const char* cut = p;
107 while (p < end) {
108 const char* prev_p = p; // Save position before decoding
109 char32_t ch = utf8_decode(p, end);
110 if (ch == 0 && p >= end) break;
111 int ch_width = 0;
112 if (is_wide_codepoint(ch)) {
113 ch_width = 2;
114 } else if (ch < 0x20 || ch == 0x7F) {
115 ch_width = 0;
116 } else {
117 ch_width = 1;
118 }
119 if (width + ch_width > max_width) break;
120 width += ch_width;
121 cut = p; // cut points to the next byte after this complete codepoint
122 }
123
124 // Ensure we don't cut in the middle of a multi-byte UTF-8 sequence
125 // If cut is not at a valid UTF-8 boundary, move it back
126 while (cut > utf8_str.data()) {
127 unsigned char c = static_cast<unsigned char>(*cut);
128 // Continuation bytes are 10xxxxxx (0x80-0xBF)
129 if ((c & 0xC0) == 0x80) {
130 cut--; // Move back to find the start of the codepoint
131 } else {
132 break; // Found a valid UTF-8 start byte or ASCII
133 }
134 }
135
136 return utf8_str.substr(0, static_cast<size_t>(cut - utf8_str.data()));
137}

151inline std::string trim(const std::string& s) {
152 if (s.empty()) return s;
153 auto start = s.begin();
154 while (start != s.end() && std::isspace(static_cast<unsigned char>(*start))) start++;
155 if (start == s.end()) return "";
156 auto end = s.end();
157 do { end--; } while (start != end && std::isspace(static_cast<unsigned char>(*end)));
158 return std::string(start, end + 1);
159}
