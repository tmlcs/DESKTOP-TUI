18 WindowsTerminal() : raw_mode_(false), alt_screen_(false) {
19 hOut_ = GetStdHandle(STD_OUTPUT_HANDLE);
20 hIn_ = GetStdHandle(STD_INPUT_HANDLE);
21
22 CONSOLE_SCREEN_BUFFER_INFO csbi;

24 cols_ = csbi.dwSize.X;

34 supports_vt_ = SetConsoleMode(hOut_, mode | ENABLE_VIRTUAL_TERMINAL_PROCESSING) != 0;
35 }

38 bool init() override {
39 // SEC-04: Validate terminal initialization on Windows
40 // Check if we have a valid console handle
41 if (hOut_ == INVALID_HANDLE_VALUE || hIn_ == INVALID_HANDLE_VALUE) {
42 // No console attached - could be a GUI app or redirected output
43 // This limits functionality but isn't necessarily fatal

45 }
46
47 // Validate minimum terminal size
48 if (cols_ < 10 || rows_ < 5) {
49 // Terminal too small for meaningful UI
50 // Could fallback to a message or refuse to initialize
51 }

138 void fill(int x, int y, int w, int h, char ch, const Style& style) override {
139 if (!supports_vt_) {
140 COORD pos = {(SHORT)x, (SHORT)y};
141 DWORD written;
142 // Map Style to Windows console attributes
143 WORD attr = FOREGROUND_RED | FOREGROUND_GREEN | FOREGROUND_BLUE; // default white
144 if (style.fg.mode == Color::Mode::TrueColor) {
145 if (style.fg.rgb.r > 128) attr |= FOREGROUND_RED;
146 if (style.fg.rgb.g > 128) attr |= FOREGROUND_GREEN;
147 if (style.fg.rgb.b > 128) attr |= FOREGROUND_BLUE;
148 }
149 if (style.bold) attr |= FOREGROUND_INTENSITY;
150 for (int row = 0; row < h; row++) {
151 FillConsoleOutputCharacterA(hOut_, ch, w, pos, &written);
152 FillConsoleOutputAttribute(hOut_, attr, w, pos, &written);
153 pos.Y++;
154 }
155 return;
156 }
157 std::string line(w, ch);
158 for (int row = y; row < y + h; row++) {
159 write_styled_at(x, row, line, style);
160 }
161 }

182 void draw_rect(int x, int y, int w, int h, const Style& style) override {
183 if (w < 2 || h < 2) return;
184 const char *tl, *tr, *bl, *br, *hline, *vline;
185 if (supports_vt_) {
186 tl = "\xe2\x94\x8c"; tr = "\xe2\x94\x90";
187 bl = "\xe2\x94\x94"; br = "\xe2\x94\x98";
188 hline = "\xe2\x94\x80"; vline = "\xe2\x94\x82";
189 } else {
190 tl = tr = bl = br = "+"; hline = "-"; vline = "|";

193 for (int i = 1; i < w - 1; i++) write_styled_at(x + i, y, hline, style);
194 write_styled_at(x + w - 1, y, tr, style);
195 for (int row = y + 1; row < y + h - 1; row++) {
196 write_styled_at(x, row, vline, style);

204 void enter_raw_mode() override {
205 if (raw_mode_) return;
206 DWORD mode = 0;
207 GetConsoleMode(hIn_, &mode);
208 orig_input_mode_ = mode;
209 // Combine raw mode flags with window/mouse input in a single call
210 SetConsoleMode(hIn_,
211 (mode & ~(ENABLE_LINE_INPUT | ENABLE_ECHO_INPUT | ENABLE_PROCESSED_INPUT))
212 | ENABLE_WINDOW_INPUT | ENABLE_MOUSE_INPUT);
213
214 GetConsoleMode(hOut_, &mode);
215 orig_output_mode_ = mode;
216 if (supports_vt_) {
217 SetConsoleMode(hOut_, mode | ENABLE_VIRTUAL_TERMINAL_PROCESSING | ENABLE_PROCESSED_OUTPUT);
218 }
219
220 if (supports_vt_) write("\033[?1006;1000;1015;2004h");
221 raw_mode_ = true;
222 }

224 void leave_raw_mode() override {
225 if (!raw_mode_) return;
226 if (supports_vt_) write("\033[?1006;1000;1015;2004l");
227 SetConsoleMode(hIn_, orig_input_mode_);

248 void set_title(const std::string& title) override {
249 // SEC-02: Sanitize title to prevent escape sequence injection
250 // Remove or replace control characters that could be used for attacks
251 std::string sanitized;
252 sanitized.reserve(title.size());
253 for (char c : title) {
254 // Allow printable ASCII and UTF-8 continuation bytes
255 // Block: ESC (0x1B), BEL (0x07), CR (0x0D), and other control chars
256 if (c == '\033' || c == '\007' || c == '\r' || (c >= 0 && c < 32 && c != '\t')) {
257 // Replace dangerous characters with space or skip them
258 if (c == '\033') {
259 // Skip ESC and any following sequence
260 continue;
261 } else if (c == '\007' || c == '\r') {
262 sanitized += ' '; // Replace with space
263 } else {
264 // Skip other control characters
265 continue;
266 }
267 } else {
268 sanitized += c;
269 }
270 }
271
272 SetConsoleTitleA(sanitized.c_str());
273 }
