56PluginResult DynamicPlugin::load() {

58 return PluginResult::ok("Plugin already loaded");

60
61 state_ = PluginState::Loading;
62
63 // Cargar la biblioteca compartida
64 // RTLD_NOW: resolver todos los símbolos inmediatamente
65 // RTLD_LOCAL: los símbolos no están disponibles para otras bibliotecas cargadas dinámicamente
66 handle_ = dlopen(path_.c_str(), RTLD_NOW | RTLD_LOCAL);
67
68 if (!handle_) {
69 std::string error = dlerror();
70 state_ = PluginState::Error;
71 return PluginResult::fail("Failed to load plugin: " + error);
72 }
73
74 // Limpiar errores previos
75 dlerror();
76
77 // Obtener función factory create_plugin
78 create_func_ = reinterpret_cast<CreateFunc>(dlsym(handle_, "create_plugin"));
79 const char* error = dlerror();
80 if (error != nullptr) {
81 dlclose(handle_);
82 handle_ = nullptr;
83 state_ = PluginState::Error;
84 return PluginResult::fail("Missing create_plugin symbol: " + std::string(error));
85 }
86
87 // Obtener función destroy_plugin
88 destroy_func_ = reinterpret_cast<DestroyFunc>(dlsym(handle_, "destroy_plugin"));
89 error = dlerror();
90 if (error != nullptr) {
91 dlclose(handle_);
92 handle_ = nullptr;
93 state_ = PluginState::Error;
94 return PluginResult::fail("Missing destroy_plugin symbol: " + std::string(error));
95 }
96
97 // Crear instancia del plugin
98 auto result = create_instance();
99 if (!result.success) {
100 dlclose(handle_);
101 handle_ = nullptr;
102 state_ = PluginState::Error;

104 }

110PluginResult DynamicPlugin::unload() {
111 if (state_ == PluginState::Unloaded || state_ == PluginState::Error) {
112 return PluginResult::ok("Plugin not loaded");

114
115 // Si está inicializado, hacer shutdown primero
116 if (state_ >= PluginState::Initialized && instance_) {

123 destroy_instance();
