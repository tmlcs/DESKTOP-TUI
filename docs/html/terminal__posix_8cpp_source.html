82 bool check_resize(int& new_cols, int& new_rows) {
83 // SEC-03: Use atomic flag instead of instance member for signal safety
84 if (!g_resize_pending.load()) return false;
85 g_resize_pending.store(false);
86 int old_cols = cols_;
87 int old_rows = rows_;
88 update_size();

90 new_rows = rows_;

207 void enter_raw_mode() override {
208 if (raw_mode_) return;
209 if (tcgetattr(STDIN_FILENO, &orig_termios_) != 0) return; // failed — don't set raw_mode_
210 struct termios raw = orig_termios_;
211 raw.c_iflag &= ~(BRKINT | ICRNL | INPCK | ISTRIP | IXON);
212 raw.c_oflag &= ~(OPOST);
213 raw.c_cflag |= (CS8);
214 raw.c_lflag &= ~(ECHO | ICANON | IEXTEN | ISIG);
215 raw.c_cc[VMIN] = 0; // non-blocking
216 raw.c_cc[VTIME] = 0;
217 tcsetattr(STDIN_FILENO, TCSAFLUSH, &raw);
218 raw_mode_ = true;
219
220 // Enable mouse (SGR 1006) and bracketed paste
221 write("\033[?1006;1000;1015;2004h");
222 flush();
223 }

245 void set_title(const std::string& title) override {
246 // SEC-02: Sanitize title to prevent escape sequence injection
247 // Remove or replace control characters that could be used for attacks
248 std::string sanitized;
249 sanitized.reserve(title.size());
250 for (char c : title) {
251 // Allow printable ASCII and UTF-8 continuation bytes
252 // Block: ESC (0x1B), BEL (0x07), CR (0x0D), and other control chars
253 if (c == '\033' || c == '\007' || c == '\r' || (c >= 0 && c < 32 && c != '\t')) {
254 // Replace dangerous characters with space or skip them
255 if (c == '\033') {
256 // Skip ESC and any following sequence
257 continue;
258 } else if (c == '\007' || c == '\r') {
259 sanitized += ' '; // Replace with space
260 } else {
261 // Skip other control characters
262 continue;
263 }
264 } else {
265 sanitized += c;
266 }
267 }
268
269 // Write the sanitized title using OSC sequence
270 // OSC 0 ; title BEL - Set icon name and window title
271 write("\033]0;" + sanitized + "\007");
272 }

290 // SEC-02: Validate dimensions to prevent underflow/overflow
291 if (ws.ws_col >= MIN_TERM_COLS && ws.ws_col <= MAX_TERM_COLS) {
292 cols_ = ws.ws_col;
293 } else if (ws.ws_col < MIN_TERM_COLS) {
294 cols_ = MIN_TERM_COLS;
