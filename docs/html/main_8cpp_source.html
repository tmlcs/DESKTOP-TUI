103 void run() {
104 renderer_.mark_dirty();
105 bool needs_render = true;
106
107 while (g_running) {
108 // Check for pending resize (signal-driven)
109 int new_cols = 0, new_rows = 0;
110 if (term_->check_resize(new_cols, new_rows)) {
111 renderer_.resize(new_cols, new_rows);
112 desktop_mgr_.on_resize(new_cols, new_rows);
113 needs_render = true;
114 }
115
116 // Process all pending input events
117 bool had_input = false;
118 while (auto event = poll_input()) {
119 if (!handle_event(*event)) break;
120 had_input = true;
121 }
122
123 // Only render if something changed
124 if (had_input || needs_render) {
125 render();
126 needs_render = false;
127 } else {
128 // Sleep when idle — configured duration gives ~20fps wake-up for input checks
129 std::this_thread::sleep_for(Config::IDLE_SLEEP_DURATION);
130 }
131 }

142 bool handle_event(const Event& e) {
143 // Global keybindings
144 if (e.type == EventType::KeyPress) {
145 // Ctrl+Q: Quit
146 if (e.mods.control && (e.key_code == 'Q' || e.key_code == 'q')) {
147 g_running = false;
148 return false;
149 }
150
151 // Alt+1..9: Switch desktop
152 if (e.mods.alt && e.key_code >= '1' && e.key_code <= '9') {
153 int idx = e.key_code - '1';
154 desktop_mgr_.switch_to(idx);
155 return true;
156 }
157
158 // Alt+Left/Right: Prev/Next desktop
159 if (e.mods.alt && e.key_code == Keys::ArrowLeft) {
160 desktop_mgr_.switch_prev();
161 return true;
162 }
163 if (e.mods.alt && e.key_code == Keys::ArrowRight) {
164 desktop_mgr_.switch_next();
165 return true;
166 }
167
168 // Alt+N: New desktop
169 if (e.mods.alt && (e.key_code == 'N' || e.key_code == 'n')) {
170 int count = desktop_mgr_.desktop_count() + 1;
171 desktop_mgr_.add_desktop("Desktop " + std::to_string(count));
172 return true;
173 }
174
175 // Alt+W: Close focused window
176 if (e.mods.alt && (e.key_code == 'W' || e.key_code == 'w')) {
177 auto* active = desktop_mgr_.active_desktop();
178 if (active) {
179 for (auto& win : active->windows()) {
180 if (win->is_focused()) {
181 active->remove_window(win->id());
182 break;
183 }
184 }
185 }
186 return true;
187 }
188
189 // Alt+Tab: Cycle windows
190 if (e.mods.alt && e.key_code == Keys::Tab) {
191 auto* active = desktop_mgr_.active_desktop();
192 if (active) {
193 bool found_focused = false;
194 Window* next_win = nullptr;
195 for (auto& win : active->windows()) {
196 if (found_focused && win->visible()) {
197 next_win = win.get();
198 break;
199 }
200 if (win->is_focused()) found_focused = true;
201 }
202 if (!next_win && !active->windows().empty()) {
203 next_win = active->windows().front().get();
204 }
205 if (next_win) {
206 for (auto& win : active->windows()) {
207 win->blur();
208 }
209 next_win->focus();
210 }
211 }
212 return true;
213 }
214
215 // Escape: minimize focused window
216 if (e.key_code == Keys::Escape) {
217 auto* active = desktop_mgr_.active_desktop();
218 if (active) {
219 for (auto& win : active->windows()) {
220 if (win->is_focused()) {
221 win->minimize();
222 break;
223 }
224 }
225 }
226 return true;
227 }
228
229 // FIX C5: Alt+R restore minimized windows
230 if (e.mods.alt && (e.key_code == 'R' || e.key_code == 'r')) {
231 auto* active = desktop_mgr_.active_desktop();
232 if (active) {
233 for (auto& win : active->windows()) {
234 if (win->is_minimized()) {
235 win->restore();
236 win->focus();
237 break;
238 }
239 }
240 }
241 return true;
242 }
243 }
244
245 // Resize events
246 if (e.type == EventType::Resize) {
247 renderer_.resize(e.cols, e.rows);
248 desktop_mgr_.on_resize(e.cols, e.rows);
249 renderer_.mark_dirty();
250 return true;
251 }
252
253 // Dispatch to active desktop's focused window
254 desktop_mgr_.dispatch_event(e);
255
256 return true;
257 }
