44 std::optional<Event> poll() override {
45 // If we have buffered bytes from a previous partial read, try to complete them
46 if (!buffer_.empty()) {
47 // Check buffer limit before reading more (prevent DoS)
48 if (buffer_.size() >= max_buffer_size_) {
49 // Buffer overflow - discard oldest data and start fresh
50 buffer_.clear();
51 Event e(EventType::KeyPress);
52 e.key_code = Keys::Escape;
53 e.key_name = "Escape";
54 return e;
55 }
56
57 // Try to read more to complete the sequence
58 unsigned char more[16];
59 ssize_t n = read(STDIN_FILENO, more, sizeof(more));
60 if (n > 0) {
61 // Ensure we don't exceed max buffer size
62 size_t available = max_buffer_size_ - buffer_.size();
63 size_t to_copy = (size_t)n < available ? (size_t)n : available;
64 buffer_.insert(buffer_.end(), more, more + to_copy);
65 }
66 // Try to parse the buffer
67 auto result = try_parse_buffer();
68 if (result.has_value()) {
69 return result;
70 }
71 // If still incomplete after adding more data, return nullopt
72 // The next poll() will try again
73 if (n > 0) return std::nullopt;
74 // No more data available, parse what we have
75 auto fallback = try_parse_buffer();
76 if (fallback.has_value()) return fallback;
77 // Buffer is stale/incomplete, clear it and return escape
78 buffer_.clear();
79 Event e(EventType::KeyPress);
80 e.key_code = Keys::Escape;
81 e.key_name = "Escape";
82 return e;
83 }
84
85 unsigned char buf[32];
86 ssize_t n = read(STDIN_FILENO, buf, sizeof(buf));
87 if (n == 0) {
88 // EOF — terminal closed, signal quit
89 Event e(EventType::Quit);
90 return e;
91 }
92 if (n < 0) return std::nullopt; // error or no data
93
94 // Try to parse; if incomplete, buffer the bytes
95 buffer_.assign(buf, buf + n);
96 auto result = try_parse_buffer();
97 if (result.has_value()) {
98 return result;
99 }
100
101 // Incomplete sequence — save to buffer for next poll
102 // This handles fragmented input
103 auto fallback = try_parse_buffer();
104 if (fallback.has_value()) return fallback;
105
106 // Truly incomplete, return escape as fallback
107 buffer_.clear();
108 Event e(EventType::KeyPress);
109 e.key_code = Keys::Escape;
110 e.key_name = "Escape";
111 return e;
112 }

120 std::optional<Event> try_parse_buffer() {
121 if (buffer_.empty()) return std::nullopt;
122
123 if (buffer_[0] == '\033') {
124 if (buffer_.size() == 1) {
125 // Incomplete: could be lone escape or start of sequence
126 return std::nullopt; // wait for more
127 }
128 if (buffer_[1] == '[') {
129 auto result = try_parse_csi();
130 if (result.has_value()) return result;
131 // CSI sequence might be incomplete
132 return std::nullopt;
133 }
134 if (buffer_[1] == 'O') {
135 auto result = try_parse_ss3();
136 if (result.has_value()) return result;
137 return std::nullopt;
138 }
139 if (buffer_[1] >= ' ' && buffer_[1] <= '~') {
140 // Alt+key (complete)
141 Event e(EventType::KeyPress);
142 e.key_code = buffer_[1];
143 e.key_name = std::string(1, (char)buffer_[1]);
144 e.mods.alt = true;
145 buffer_.erase(buffer_.begin(), buffer_.begin() + 2);
146 return e;
147 }
148 // Unknown escape sequence, skip it
149 buffer_.erase(buffer_.begin());
150 return std::nullopt;
151 }
152
153 // Regular character
154 Event e(EventType::KeyPress);
155 e.key_code = buffer_[0];
156 if (buffer_[0] == '\r' || buffer_[0] == '\n') {
157 e.key_code = Keys::Enter;
158 e.key_name = "Enter";
159 } else if (buffer_[0] == '\t') {
160 e.key_code = Keys::Tab;
161 e.key_name = "Tab";
162 } else if (buffer_[0] == '\b' || buffer_[0] == 127) {
163 e.key_code = Keys::Backspace;
164 e.key_name = "Backspace";
165 } else if (buffer_[0] < 32) {
166 e.key_code = buffer_[0] + '@';
167 e.mods.control = true;
168 e.key_name = std::string(1, (char)e.key_code);
169 } else {
170 e.key_name = std::string(1, (char)buffer_[0]);
171 }
172 buffer_.erase(buffer_.begin());
173 return e;
174 }

176 std::optional<Event> try_parse_csi() {
177 // Minimum CSI: ESC [ X = 3 bytes
178 if (buffer_.size() < 3) return std::nullopt;
179
180 // SEC-01: Handle bracketed paste (ESC [ 2 0 0 ~ starts paste, ESC [ 2 0 1 ~ ends)
181 // We treat all content between these markers as raw text, not commands
182 if (buffer_.size() >= 6 && buffer_[2] == '2' && buffer_[3] == '0') {
183 // Check for start (200~) or end (201~) of bracketed paste
184 if (buffer_.size() >= 7 && buffer_[4] == '0' && buffer_[5] == '~') {
185 // Start of bracketed paste: ESC[200~
186 // Clear buffer and mark that we're in paste mode
187 buffer_.erase(buffer_.begin(), buffer_.begin() + 6);
188 in_bracketed_paste_ = true;
189 return std::nullopt; // Consume the sequence, wait for paste content
190 }
191 if (buffer_.size() >= 7 && buffer_[4] == '1' && buffer_[5] == '~') {
192 // End of bracketed paste: ESC[201~
193 buffer_.erase(buffer_.begin(), buffer_.begin() + 6);
194 in_bracketed_paste_ = false;
195 return std::nullopt; // Consume the sequence
196 }
197 }
198
199 // If we're inside bracketed paste, treat all input as raw text
200 if (in_bracketed_paste_) {
201 // Look for the end sequence ESC[201~
202 size_t end_pos = 0;
203 bool found_end = false;
204 for (size_t i = 0; i + 6 <= buffer_.size(); i++) {
205 if (buffer_[i] == '\033' && buffer_[i+1] == '[' &&
206 buffer_[i+2] == '2' && buffer_[i+3] == '0' &&
207 buffer_[i+4] == '1' && buffer_[i+5] == '~') {
208 end_pos = i;
209 found_end = true;
210 break;
211 }
212 }
213
214 if (found_end) {
215 // Extract paste content (everything before ESC[201~)
216 std::string paste_content(buffer_.begin(), buffer_.begin() + static_cast<ptrdiff_t>(end_pos));
217 // Remove the entire sequence including end marker
218 buffer_.erase(buffer_.begin(), buffer_.begin() + static_cast<ptrdiff_t>(end_pos) + 6);
219
220 // Create a custom event with the paste content
221 Event e(EventType::Custom);
222 e.data_s = paste_content;
223 e.key_name = "BracketedPaste";
224 in_bracketed_paste_ = false;
225 return e;
226 } else {
227 // Incomplete paste - wait for more data
228 return std::nullopt;
229 }
230 }
231
232 // SGR 1006 mouse: ESC [ < button ; x ; y M/m
233 if (buffer_.size() >= 4 && buffer_[2] == '<') {
234 // Try to find the terminator ('M' or 'm')
235 size_t term_pos = 0;
236 bool found = false;
237 for (size_t i = 3; i < buffer_.size(); i++) {
238 if (buffer_[i] == 'M' || buffer_[i] == 'm') {
239 term_pos = i;
240 found = true;
241 break;
242 }
243 }
244 if (!found) return std::nullopt; // incomplete
245
246 Event e;
247 int button, x, y;
248 char last;
249 std::string seq(buffer_.begin() + 3, buffer_.begin() + term_pos + 1);
250 if (sscanf(seq.c_str(), "%d;%d;%d%c", &button, &x, &y, &last) == 4) {
251 mouse_x_ = x - 1;
252 mouse_y_ = y - 1;
253
254 if (last == 'M') {
255 // Scroll wheel: buttons 64=up, 65=down, 66=left, 67=right
256 if (button >= 64 && button <= 67) {
257 e.type = EventType::MouseScroll;
258 switch (button) {
259 case 64: e.scroll_delta = -1; break; // scroll up
260 case 65: e.scroll_delta = 1; break; // scroll down
261 case 66: e.scroll_delta = -2; break; // scroll left
262 case 67: e.scroll_delta = 2; break; // scroll right
263 }
264 } else {
265 e.type = EventType::MouseDown;
266 e.mouse_button = button & 3;
267 }
268 e.mods.shift = button & 4;
269 e.mods.meta = button & 8;
270 e.mods.control = button & 16;
271 } else if (last == 'm') {
272 e.type = EventType::MouseUp;
273 e.mouse_button = button & 3;
274 } else {
275 e.type = EventType::MouseMove;
276 e.mouse_button = button & 3;
277 }
278 e.mouse_x = mouse_x_;
279 e.mouse_y = mouse_y_;
280 buffer_.erase(buffer_.begin(), buffer_.begin() + static_cast<ptrdiff_t>(term_pos) + 1);
281 return e;
282 }
283 // Parse failed, skip this sequence
284 buffer_.erase(buffer_.begin(), buffer_.begin() + static_cast<ptrdiff_t>(term_pos) + 1);
285 return std::nullopt;
286 }
287
288 // Arrow keys: ESC [ A/B/C/D = 3 bytes
289 if (buffer_.size() >= 3 && buffer_[2] >= 'A' && buffer_[2] <= 'D') {
290 Event e;
291 e.type = EventType::KeyPress;
292 switch (buffer_[2]) {
293 case 'A': e.key_code = Keys::ArrowUp; e.key_name = "ArrowUp"; break;
294 case 'B': e.key_code = Keys::ArrowDown; e.key_name = "ArrowDown"; break;
295 case 'C': e.key_code = Keys::ArrowRight; e.key_name = "ArrowRight"; break;
296 case 'D': e.key_code = Keys::ArrowLeft; e.key_name = "ArrowLeft"; break;
297 }
298 buffer_.erase(buffer_.begin(), buffer_.begin() + 3);
299 return e;
300 }
301
302 // Home/End/Shift-Tab: ESC [ H/F/Z = 3 bytes
303 if (buffer_.size() >= 3) {
304 switch (buffer_[2]) {
305 case 'H': {
306 Event e; e.type = EventType::KeyPress; e.key_code = Keys::Home; e.key_name = "Home";
307 buffer_.erase(buffer_.begin(), buffer_.begin() + 3); return e;
308 }
309 case 'F': {
310 Event e; e.type = EventType::KeyPress; e.key_code = Keys::End; e.key_name = "End";
311 buffer_.erase(buffer_.begin(), buffer_.begin() + 3); return e;
312 }
313 case 'Z': {
314 Event e; e.type = EventType::KeyPress; e.key_code = Keys::Tab; e.key_name = "Tab";
315 e.mods.shift = true;
316 buffer_.erase(buffer_.begin(), buffer_.begin() + 3); return e;
317 }
318 }
319 }
320
321 // Extended: ESC [ 1 ; 2 ~ etc (need at least 6 bytes for "ESC[1;2~")
322 // ESC [ N ~ where N is a number
323 if (buffer_.size() >= 4 && buffer_[2] >= '0' && buffer_[2] <= '9') {
324 // Find the terminator '~'
325 size_t term_pos = 0;
326 bool found = false;
327 for (size_t i = 3; i < buffer_.size(); i++) {
328 if (buffer_[i] == '~') {
329 term_pos = i;
330 found = true;
331 break;
332 }
333 }
334 if (!found) return std::nullopt; // incomplete
335
336 int code;
337 std::string num_str(buffer_.begin() + 2, buffer_.begin() + static_cast<ptrdiff_t>(term_pos));
338 if (sscanf(num_str.c_str(), "%d", &code) == 1) {
339 Event e;
340 e.type = EventType::KeyPress;
341 switch (code) {
342 case 1: case 7: e.key_code = Keys::Home; e.key_name = "Home"; break;
343 case 2: e.key_code = Keys::Insert; e.key_name = "Insert"; break;
344 case 3: e.key_code = Keys::Delete_; e.key_name = "Delete"; break;
345 case 4: case 8: e.key_code = Keys::End; e.key_name = "End"; break;
346 case 5: e.key_code = Keys::PageUp; e.key_name = "PageUp"; break;
347 case 6: e.key_code = Keys::PageDown; e.key_name = "PageDown"; break;
348 case 11: e.key_code = Keys::F1; e.key_name = "F1"; break;
349 case 12: e.key_code = Keys::F2; e.key_name = "F2"; break;
350 case 13: e.key_code = Keys::F3; e.key_name = "F3"; break;
351 case 14: e.key_code = Keys::F4; e.key_name = "F4"; break;
352 case 15: e.key_code = Keys::F5; e.key_name = "F5"; break;
353 case 17: e.key_code = Keys::F6; e.key_name = "F6"; break;
354 case 18: e.key_code = Keys::F7; e.key_name = "F7"; break;
355 case 19: e.key_code = Keys::F8; e.key_name = "F8"; break;
356 case 20: e.key_code = Keys::F9; e.key_name = "F9"; break;
357 case 21: e.key_code = Keys::F10; e.key_name = "F10"; break;
358 default: e.key_code = buffer_[0]; e.key_name = "Unknown"; break;
359 }
360 buffer_.erase(buffer_.begin(), buffer_.begin() + static_cast<ptrdiff_t>(term_pos) + 1);
361 return e;
362 }
363 buffer_.erase(buffer_.begin(), buffer_.begin() + static_cast<ptrdiff_t>(term_pos) + 1);
364 return std::nullopt;
365 }
366
367 return std::nullopt; // incomplete or unrecognized
368 }

372 if (buffer_.size() < 3) return std::nullopt;

374 Event e;

379 case 'R': e.key_code = Keys::F3; e.key_name = "F3"; break;
380 case 'S': e.key_code = Keys::F4; e.key_name = "F4"; break;
381 case 'H': e.key_code = Keys::Home; e.key_name = "Home"; break;
382 case 'F': e.key_code = Keys::End; e.key_name = "End"; break;
383 default: return std::nullopt;
384 }
385 buffer_.erase(buffer_.begin(), buffer_.begin() + 3);
386 return e;
