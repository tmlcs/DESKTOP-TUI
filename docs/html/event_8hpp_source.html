117class EventBus {
118public:
119 using SubscriptionId = uint64_t;
120
121 SubscriptionId subscribe(EventType type, EventHandler handler) {
122 auto id = next_id_++;
123 handlers_[type].push_back({id, std::move(handler)});
124 return id;
125 }
126
127 void unsubscribe(SubscriptionId id) {
128 for (auto& [type, vec] : handlers_) {
129 vec.erase(
130 std::remove_if(vec.begin(), vec.end(),
131 [id](const auto& h) { return h.id == id; }),
132 vec.end()
133 );
134 }
135 }
136
137 void publish(const Event& event) {
138 auto it = handlers_.find(event.type);
139 if (it != handlers_.end()) {
140 // FIX C7: snapshot to prevent iterator invalidation
141 auto snapshot = it->second;
142 for (const auto& [id, handler] : snapshot) {
143 handler(event);
144 }
145 }
146 // Also publish to wildcard handlers
147 if (!wildcard_handlers_.empty()) {
148 // FIX C7: snapshot wildcard handlers too
149 auto snapshot = wildcard_handlers_;
150 for (const auto& handler : snapshot) {
151 handler.callback(event);
152 }
153 }
154 }
155
156 SubscriptionId subscribe_all(EventHandler handler) {
157 auto id = next_id_++;
158 wildcard_handlers_.push_back({id, std::move(handler)});
159 return id;
160 }
161
162private:
163 struct Handler {
164 SubscriptionId id;
165 EventHandler callback;
166 };
167
168 uint64_t next_id_ = 1;
169 std::map<EventType, std::vector<Handler>> handlers_;
170 std::vector<Handler> wildcard_handlers_;
171};
