24 std::optional<Event> poll() override {
25 int ch = getchar();
26 if (ch == EOF) return std::nullopt;
27
28 Event e(EventType::KeyPress);
29 e.key_code = ch;
30
31 if (ch == '\033') {
32 // Check for escape sequence
33 int ch2 = getchar();
34 if (ch2 == EOF) {
35 e.key_code = Keys::Escape;
36 e.key_name = "Escape";
37 return e;
38 }
39 if (ch2 == '[') {
40 int ch3 = getchar();
41 if (ch3 >= 'A' && ch3 <= 'D') {
42 switch (ch3) {
43 case 'A': e.key_code = Keys::ArrowUp; e.key_name = "ArrowUp"; break;
44 case 'B': e.key_code = Keys::ArrowDown; e.key_name = "ArrowDown"; break;
45 case 'C': e.key_code = Keys::ArrowRight; e.key_name = "ArrowRight"; break;
46 case 'D': e.key_code = Keys::ArrowLeft; e.key_name = "ArrowLeft"; break;
47 }
48 return e;
49 }
50 }
51 // Alt+key
52 e.mods.alt = true;
53 e.key_code = ch2;
54 e.key_name = std::string(1, (char)ch2);
55 return e;
56 }
57
58 if (ch == '\r' || ch == '\n') {
59 e.key_code = Keys::Enter; e.key_name = "Enter";
60 } else if (ch == '\t') {
61 e.key_code = Keys::Tab; e.key_name = "Tab";
62 } else if (ch == '\b' || ch == 127) {
63 e.key_code = Keys::Backspace; e.key_name = "Backspace";
64 } else if (ch >= 32) {
65 e.key_name = std::string(1, (char)ch);
66 }
67
68 return e;
69 }
