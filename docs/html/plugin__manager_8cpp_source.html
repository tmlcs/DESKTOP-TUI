106PluginResult PluginManager::shutdown() {
107 std::lock_guard<std::mutex> lock(mutex_);
108
109 if (!initialized_) {
110 return PluginResult::ok("PluginManager not initialized");
111 }
112
113 PluginResult overall_result = PluginResult::ok();
114
115 // Hacer shutdown de todos los plugins en orden inverso
116 std::vector<std::string> plugin_names;
117 for (const auto& [name, _] : plugins_) {
118 plugin_names.push_back(name);
119 }
120
121 std::reverse(plugin_names.begin(), plugin_names.end());
122
123 for (const auto& name : plugin_names) {
124 auto result = unload_plugin(name);
125 if (!result.success) {
126 overall_result = PluginResult::fail(
127 "Failed to unload plugin " + name + ": " + result.message
128 );
129 }
130 }
131
132 plugins_.clear();
133 callbacks_.clear();
134 config_.clear();
135 widgets_.clear();
136 context_.reset();
137
138 initialized_ = false;
139
140 return overall_result;

143PluginResult PluginManager::load_plugin(const std::string& path) {
144 std::lock_guard<std::mutex> lock(mutex_);
145
146 if (!initialized_) {
147 return PluginResult::fail("PluginManager not initialized");
148 }
149
150 // Verificar que el archivo existe
151 if (!fs::exists(path)) {
152 return PluginResult::fail("Plugin file does not exist: " + path);
153 }
154
155 // Extraer nombre del plugin
156 std::string name = extract_plugin_name(path);
157
158 // Verificar si ya está cargado
159 if (plugins_.find(name) != plugins_.end()) {
160 return PluginResult::ok("Plugin already loaded: " + name);
161 }
162
163 // Crear wrapper dinámico
164 auto plugin = std::make_unique<DynamicPlugin>(path);
165
166 // Cargar el plugin
167 auto load_result = plugin->load();
168 if (!load_result.success) {
169 return load_result;
170 }
171
172 // Validar dependencias
173 auto info = plugin->get_info();
174 auto dep_result = validate_dependencies(info);
175 if (!dep_result.success) {
176 plugin->unload();
177 return dep_result;
178 }
179
180 // Inicializar el plugin con el contexto
181 auto init_result = plugin->get_instance()->initialize(*context_);
182 if (!init_result.success) {
183 plugin->unload();
184 return PluginResult::fail("Failed to initialize plugin: " + init_result.message);
185 }
186
187 // Registrar el plugin
188 plugins_[name] = std::move(plugin);
189
190 return PluginResult::ok("Plugin loaded and initialized: " + name + " (" + info.to_string() + ")");
191}
