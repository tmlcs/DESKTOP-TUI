27 INPUT_RECORD record;
28 DWORD read = 0;
29 ReadConsoleInput(hIn_, &record, 1, &read);
30 if (read == 0) return std::nullopt;
31
32 if (record.EventType == KEY_EVENT && record.Event.KeyEvent.bKeyDown) {

35 if (record.EventType == MOUSE_EVENT) {

101 std::optional<Event> process_mouse(const MOUSE_EVENT_RECORD& mer) {
102 Event e;
103 mouse_x_ = mer.dwMousePosition.X;
104 mouse_y_ = mer.dwMousePosition.Y;
105 e.mouse_x = mouse_x_;
106 e.mouse_y = mouse_y_;
107
108 if (mer.dwEventFlags == MOUSE_MOVED) {
109 e.type = EventType::MouseMove;
110 return e;
111 }
112 if (mer.dwEventFlags == MOUSE_WHEELED) {
113 e.type = EventType::MouseScroll;
114 e.scroll_delta = (mer.dwButtonState >> 16) > 0 ? 1 : -1;
115 return e;
116 }
117
118 e.type = EventType::MouseDown;
119 if (mer.dwButtonState & FROM_LEFT_1ST_BUTTON_PRESSED) e.mouse_button = 0;
120 else if (mer.dwButtonState & FROM_LEFT_2ND_BUTTON_PRESSED) e.mouse_button = 1;
121 else if (mer.dwButtonState & RIGHTMOST_BUTTON_PRESSED) e.mouse_button = 2;
122
123 return e;
124 }
